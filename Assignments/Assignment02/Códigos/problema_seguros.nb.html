# --- 4) Tabla de IRR (Incidence Rate Ratios) ---
irr_table <- function(fit){
  est <- coef(fit)
  se  <- sqrt(diag(vcov(fit)))
  tibble(
    term = names(est),
    IRR  = exp(est),
    `2.5%` = exp(est - 1.96*se),
    `97.5%` = exp(est + 1.96*se),
    pval = summary(fit)$coef[,4]
  )
}
